#必須項目
#１.会話数  ：最初に「NUM_TALK」を書き、「=」で区切って、会話数を書く。実際の会話数とずれるとエラーになります。
#２.会話内容：最初に「SET_TALK」、最後は「END_TALK」で締める。（改行必須）
#             間に他の情報を設定する。　書き方は「設定する項目名 = 内容」
#             【使用できる項目名】
#             「TALK」：会話内容
#             「PLAYER」：プレイヤー番号  0：1P  1:2P  2:両プレイヤー
#             「TEXTURE」：テクスチャ番号
#
#
#
#
#********************************

#会話数
NUM_TALK = 2

#会話内容
SET_TALK
	TALK = 初めての会話
	PLAYER = 0
END_TALK

#会話内容
SET_TALK
	TALK = 2回目のお話
	PLAYER = 1
END_TALK
